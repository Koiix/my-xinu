19  csrptr = (struct eth_q_csreg *)ethptr->csr;
20 
21  /* If there is no interrupt pending, return */
22 
23  if((csrptr->sr & ETH_QUARK_SR_NIS) == 0) {
24  return;
25  }
26 
27  /* Check status register to figure out the source of interrupt */
28 
29  if (csrptr->sr & ETH_QUARK_SR_TI) { /* Transmit interrupt */
30

32 
33  csrptr->sr = ETH_QUARK_SR_TI;
34 
35  /* Pointer to the head of transmit desc ring */
36 
37  tdescptr = (struct eth_q_tx_desc *)ethptr->txRing +
38  ethptr->txHead;
39 
40  count = 0; /* Start packet count at zero */
41 
42  /* Repeat until we process all the descriptor slots */
43 
44  while(ethptr->txHead != ethptr->txTail) {
45 
46  /* If the descriptor is owned by DMA, stop here */
47 
48  if(tdescptr->ctrlstat & ETH_QUARK_TDCS_OWN) {
49  break;
50  }
51 
52  /* Descriptor was processed; increment count */
53  count++;
54 
55  /* Go to the next descriptor */
56 
57  tdescptr += 1;
58 
59  /* Increment the head of the transmit desc ring */
60 
61  ethptr->txHead += 1;
62  if(ethptr->txHead >= ethptr->txRingSize) {
63  ethptr->txHead = 0;
64  tdescptr = (struct eth_q_tx_desc *)
65  ethptr->txRing;
66  }
67  }
68 
69  /* 'count' packets were processed by DMA, and slots are */
70  /* now free; signal the semaphore accordingly */
71 
72  csrptr->sr = ETH_QUARK_SR_NIS;
73  signaln(ethptr->osem, count);
74  return;
75  } else if(csrptr->sr & ETH_QUARK_SR_RI) { /* Receive interrupt */
76 
77  /* Acknowledge the interrupt */
78 
79  csrptr->sr = ETH_QUARK_SR_RI;
80 
81  /* Get the pointer to the tail of the receive desc list */
82 
83  rdescptr = (struct eth_q_rx_desc *)ethptr->rxRing +
84  ethptr->rxTail;
85 
86  count = 0; /* Start packet count at zero */
87 
88  /* Repeat until we have received */
89  /* maximum no. packets that can fit in queue */
90 
91  while(count <= ethptr->rxRingSize) {
92 
93  /* If the descriptor is owned by the DMA, stop */
94 
95  if(rdescptr->status & ETH_QUARK_RDST_OWN) {
96  break;
97  }
98 
99  /* Descriptor was processed; increment count */
100  count++;
101 
102  /* Go to the next descriptor */
103 
104  rdescptr += 1;
105 
106  /* Increment the tail index of the rx desc ring */
107 
108  ethptr->rxTail += 1;
109  if(ethptr->rxTail >= ethptr->rxRingSize) {
110  ethptr->rxTail = 0;
111  rdescptr = (struct eth_q_rx_desc *)
112  ethptr->rxRing;
113  }
114  }
115 
116  /* 'count' packets were received and are available, */
117  /* so signal the semaphore accordingly */
118 
119  signaln(ethptr->isem, count);
120  }
121 
122  /* Acknowledge the normal interrupt summary */
123 
124  csrptr->sr = ETH_QUARK_SR_NIS;
125
