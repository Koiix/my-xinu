74 void intel_cln_early_sb_read_reg(cln_sb_id id, unsigned char cmd, unsigned char reg, uint32 *data)
75 {
76  uint32 msg = (cmd << INTEL_CLN_SB_MCR_SHIFT) |
77  ((id << INTEL_CLN_SB_PORT_SHIFT) & 0xFF0000)|
78  ((reg << INTEL_CLN_SB_REG_SHIFT) & 0xFF00)|
79  INTEL_CLN_SB_BYTEEN;
80 
81  if (data == NULL) {
82  return;
83  }
84 
85  pci_write_config_dword(sb_pcidev, INTEL_CLN_SB_CMD_ADDR, msg);
86  pci_read_config_dword(sb_pcidev, INTEL_CLN_SB_DATA_ADDR, data);
87 
88 }

187 int remove_irm_protections(void)
188 {
189  uint32 tmp_addr;
190 
191  if (intel_cln_early_sb_probe() != 0) {
192  kprintf("%s() error probing for IRM device\n", __FUNCTION__);
193  /* Critial error - bail out */
194  return SYSERR;
195  }
196 
197  intel_cln_early_sb_read_reg(SB_ID_ESRAM, CFG_READ_OPCODE, DRAM_IMR3L, &tmp_addr);
198  if (tmp_addr & IMR_LOCK_BIT) {
199  return SYSERR;
200  }
201 
202  if (tmp_addr) {
203  intel_cln_early_sb_write_reg(SB_ID_ESRAM, CFG_WRITE_OPCODE, DRAM_IMR3RM, IMR_READ_ENABLE_ALL);
204  intel_cln_early_sb_write_reg(SB_ID_ESRAM, CFG_WRITE_OPCODE, DRAM_IMR3WM, IMR_WRITE_ENABLE_ALL);
205  }
206 
207  cln_remove_imr_boot_params();
208  cln_remove_imr_bzimage();
209  cln_remove_imr_grub();
210 
211  //kprintf("%s() complete\n", __FUNCTION__);
212  return OK;
213 }
