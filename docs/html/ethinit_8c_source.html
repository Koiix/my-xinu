12  volatile struct eth_q_csreg *csrptr, /* CSR address */
13  uint32 regnum /* Register */
14  )

16  uint32 retries; /* No. of retries for read */
17 
18  /* Wait for the MII to be ready */
19 
20  while(csrptr->gmiiar & ETH_QUARK_GMIIAR_GB);
21 
22  /* Prepare the GMII address register for read transaction */
23 
24  csrptr->gmiiar =
25  (1 << 11) | /* Physical Layer Address = 1 */
26  (regnum << 6) | /* PHY Register Number */
27  (ETH_QUARK_GMIIAR_CR) | /* GMII Clock Range 100-150MHz*/
28  (ETH_QUARK_GMIIAR_GB); /* Start the transaction */
29 
30  /* Wait for the transaction to complete */
31 
32  retries = 0;
33  while(csrptr->gmiiar & ETH_QUARK_GMIIAR_GB) {
34  DELAY(ETH_QUARK_INIT_DELAY);
35  if((++retries) > ETH_QUARK_MAX_RETRIES)
36  return 0;
37  }
38 
39  /* Transaction is complete, read the PHY register value from */
40  /* the GMII data register */
41  return (uint16)csrptr->gmiidr;

85  *------------------------------------------------------------------------
86  */
87 int32 eth_phy_reset (
88  volatile struct eth_q_csreg *csrptr /* CSR address */
89  )
90 {
91  uint16 value; /* Variable to read in PHY registers */
92  uint32 retries; /* No. of retries for reset */
93 
94  /* Read the PHY control register (register 0) */
95 
96  value = eth_phy_read(csrptr, 0);
97 
98  /* Set bit 15 in control register to reset the PHY */
99 
100  eth_phy_write(csrptr, 0, (value | 0x8000));
101 
102  /* Wait for PHY reset process to complete */
103 
104  retries = 0;
105  while(eth_phy_read(csrptr, 0) & 0x8000) {
106  DELAY(ETH_QUARK_INIT_DELAY);
107  if((++retries) > ETH_QUARK_MAX_RETRIES)
108  return SYSERR;
109  }
110 
111  /* See if the PHY has auto-negotiation capability */
112 
113  value = eth_phy_read(csrptr, 1); /* PHY Status register */
114  if(value & 0x0008) { /* Auto-negotiation capable */
115 
116  /* Wait for the auto-negotiation process to complete */
117 
118  retries = 0;
119  while((eth_phy_read(csrptr, 1) & 0x0020) == 0) {
120  DELAY(ETH_QUARK_INIT_DELAY);
121  if((++retries) > ETH_QUARK_MAX_RETRIES)
122  return SYSERR;
123  }
124  }
125 
126  /* Wait for the Link to be Up */
127 
128  retries = 0;
129  while((eth_phy_read(csrptr, 1) & 0x0004) == 0) {
130  DELAY(ETH_QUARK_INIT_DELAY);
131  if((++retries) > ETH_QUARK_MAX_RETRIES)
132  return SYSERR;
133  }
134 
135  DELAY(ETH_QUARK_INIT_DELAY);
136 
137  kprintf("Ethernet Link is Up\n");
138 
139  return OK;
