59  * meminit - initialize memory bounds and the free memory list
60  *------------------------------------------------------------------------
61  */
62 void meminit(void) {
63 
64  struct memblk *memptr; /* Ptr to memory block */
65  struct mbmregion *mmap_addr; /* Ptr to mmap entries */
66  struct mbmregion *mmap_addrend; /* Ptr to end of mmap region */
67  struct memblk *next_memptr; /* Ptr to next memory block */
68  uint32 next_block_length; /* Size of next memory block */
69 
70  mmap_addr = (struct mbmregion*)NULL;
71  mmap_addrend = (struct mbmregion*)NULL;
72 
73  /* Initialize the free list */
74  memptr = &memlist;
75  memptr->mnext = (struct memblk *)NULL;
76  memptr->mlength = 0;
77 
78  /* Initialize the memory counters */
79  /* Heap starts at the end of Xinu image */
80  minheap = (void*)&end;
81  maxheap = minheap;
82 
83  /* Check if Xinu was loaded using the multiboot specification */
84  /* and a memory map was included */
85  if(bootsign != MULTIBOOT_SIGNATURE) {
86  panic("could not find multiboot signature");
87  }
88  if(!(bootinfo->flags & MULTIBOOT_BOOTINFO_MMAP)) {
89  panic("no mmap found in boot info");
90  }
91 
92  /* Get base address of mmap region (passed by GRUB) */
93  mmap_addr = (struct mbmregion*)bootinfo->mmap_addr;
94 
95  /* Calculate address that follows the mmap block */
96  mmap_addrend = (struct mbmregion*)((uint8*)mmap_addr + bootinfo->mmap_length);
97 
98  /* Read mmap blocks and initialize the Xinu free memory list */
99  while(mmap_addr < mmap_addrend) {
100 
101  /* If block is not usable, skip to next block */
102  if(mmap_addr->type != MULTIBOOT_MMAP_TYPE_USABLE) {
103  mmap_addr = (struct mbmregion*)((uint8*)mmap_addr + mmap_addr->size + 4);
104  continue;
105  }
106 
107  if((uint32)maxheap < ((uint32)mmap_addr->base_addr + (uint32)mmap_addr->length)) {
108  maxheap = (void*)((uint32)mmap_addr->base_addr + (uint32)mmap_addr->length);
109  }
110 
111  /* Ignore memory blocks within the Xinu image */
112  if((mmap_addr->base_addr + mmap_addr->length) < ((uint32)minheap)) {
113  mmap_addr = (struct mbmregion*)((uint8*)mmap_addr + mmap_addr->size + 4);
114  continue;
115  }
116 
117  /* The block is usable, so add it to Xinu's memory list */
118 
119  /* This block straddles the end of the Xinu image */
120  if((mmap_addr->base_addr <= (uint32)minheap) &&
121  ((mmap_addr->base_addr + mmap_addr->length) >
122  (uint32)minheap)) {
123 
124  /* This is the first free block, base address is the minheap */
125  next_memptr = (struct memblk *)roundmb(minheap);
126 
127  /* Subtract Xinu image from length of block */
128  next_block_length = (uint32)truncmb(mmap_addr->base_addr + mmap_addr->length - (uint32)minheap);
129  } else {
130 
131  /* Handle a free memory block other than the first one */
132  next_memptr = (struct memblk *)roundmb(mmap_addr->base_addr);
133 
134  /* Initialize the length of the block */
135  next_block_length = (uint32)truncmb(mmap_addr->length);
136  }
137 
138  /* Add then new block to the free list */
139  memptr->mnext = next_memptr;
140  memptr = memptr->mnext;
141  memptr->mlength = next_block_length;
142 
143  /* Move to the next mmap block */
144  mmap_addr = (struct mbmregion*)((uint8*)mmap_addr + mmap_addr->size + 4);
145  }
146 
147  /* End of all mmap blocks, and so end of Xinu free list */
148  if(memptr) {
149  memptr->mnext = (struct memblk *)NULL;
150  }
151 }

173  psd->sd_lolimit = ds_end;
174  psd->sd_hilim_fl = FLAGS_SETTINGS | ((ds_end >> 16) & 0xff);

177  psd->sd_lolimit = ds_end;
178  psd->sd_hilim_fl = FLAGS_SETTINGS | ((ds_end >> 16) & 0xff);

181  psd->sd_lolimit = ds_end; /* Allows execution of 0x100000 CODE */
182  psd->sd_hilim_fl = FLAGS_SETTINGS | ((ds_end >> 16) & 0xff);
