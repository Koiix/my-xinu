148 devcall sdmcread (
149  struct dentry *devptr, /* entry in device switch table */
150  char *buff, /* buffer to hold disk block */
151  int32 blk /* block number of block to read*/
152  )
153 {
154  volatile struct sdmc_csreg *csrptr; /* address of SD controller's CSR */
155  struct sdmcblk *sdmcptr; /* Pointer to sdmctab entry */
156  uint16 error_sts = 0; /* SDMC command error status */
157  uint32 cmd_arg = 0;
158 
159  /* Initialize structure pointers */
160  sdmcptr = &sdmctab[devptr->dvminor];
161  csrptr = (struct sdmc_csreg *) devptr->dvcsr;
162 
163  /* Retrieve the current card status to ensure it is ready for I/O */
164  cmd_arg = sdmcptr->rca;
165  if(sdmc_issue_cmd_sync(csrptr, SDMC_CMD13, cmd_arg, &error_sts, SDMC_CMD_NO_FLAGS) != SDMC_RC_OK) {
166  kprintf("[SDMC] Error in CMD13: %04X %04X", SDMC_CMD13, error_sts);
167  return SYSERR;
168  }
169 
170  switch(csrptr->response0 & SDMC_R1_CURRENT_STATE) {
171 
172  /* Card is in initial state the device */
173  /* must be reopened to allow I/O */
174  case SDMC_R1_IDLE_STATE:
175  case SDMC_R1_READY_STATE:
176  case SDMC_R1_IDENT_STATE:
177  /* TODO close and reopen the device */
178  return SYSERR;
179  break;
180 
181  /* Standby state, issue CMD7 to select the card */
182  case SMDC_R1_STBY_STATE:
183  cmd_arg = sdmcptr->rca;
184  if(sdmc_issue_cmd_sync(csrptr, SDMC_CMD7, cmd_arg, &error_sts, SDMC_CMD_NO_FLAGS) != SDMC_RC_OK) {
185  kprintf("[SDMC] Error in CMD7: %04X %04X", SDMC_CMD7, error_sts);
186  return SYSERR;
187  }
188  break;
189 
190  /* Card is already in data transfer state */
191  case SDMC_R1_TRAN_STATE:
192  case SDMC_R1_DATA_STATE:
193  case SDMC_R1_RCV_STATE:
194  break;
195 
196  /* Card is in some other state */
197  /* must be reopened to allow I/O */
198  case SDMC_R1_PRG_STATE:
199  case SMDC_R1_DIS_STATE:
200  /* TODO close and reopen the device */
201  return SYSERR;
202  break;
203 
204  default:
205  return SYSERR;
206  }
207 
208  sdmcread_dma(devptr, buff, blk);
209 
210  return OK;
211 }
