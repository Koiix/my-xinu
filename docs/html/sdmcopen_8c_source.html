10 devcall sdmc_set_bus_power (
11  volatile struct sdmc_csreg *csrptr /* address of SD controller's CSR */
12  )
13 {
14  uint8 sdmc_voltage_select = 0;
15 
16  /* Determine max supported voltage */
17  if(csrptr->capabilities & SDMC_CAP_VOLT_SUPPORT_3P3V) {
18  sdmc_voltage_select = SDMC_PWR_CTL_SD_BUS_VOL_SEL_3P3V;
19  } else if(csrptr->capabilities & SDMC_CAP_VOLT_SUPPORT_3P0V) {
20  sdmc_voltage_select = SDMC_PWR_CTL_SD_BUS_VOL_SEL_3P0V;
21  } else if(csrptr->capabilities & SDMC_CAP_VOLT_SUPPORT_1P8V) {
22  sdmc_voltage_select = SDMC_PWR_CTL_SD_BUS_VOL_SEL_1P8V;
23  } else {
24  /* Unknown supported voltage */
25  return SDMC_RC_NON_RECOVERABLE_ERROR;
26  }
27 
28  /* Set the controller voltage */
29  csrptr->pwr_ctl &= SDMC_PWR_CTL_SD_BUS_VOL_SEL_CLR;
30  csrptr->pwr_ctl |= sdmc_voltage_select;
31 
32  /* Set bus power flag */
33  csrptr->pwr_ctl |= SDMC_PWR_CTL_SD_BUS_PWR;
34 
35  return SDMC_RC_OK;
36 }

39  * sdmc_set_clock - set SDCLK frequency
40  *------------------------------------------------------------------------
41  */
42 devcall sdmc_set_clock (
43  volatile struct sdmc_csreg *csrptr /* address of SD controller's CSR */
44  )
45 {
46  /* Change to the SD Clock is not allowed while command inhibit is set */
47  while(csrptr->pre_state & SDMC_PRE_STATE_CMD_INHIBIT_CMD) {
48  DELAY(SDMC_CMD_DELAY);
49  }
50  while(csrptr->pre_state & SDMC_PRE_STATE_CMD_INHIBIT_DAT) {
51  DELAY(SDMC_CMD_DELAY);
52  }
53 
54  /* Disable the SD Clock */
55  csrptr->clk_ctl &= ~(SDMC_CLK_CTL_SD_CLK_EN);
56 
57  /* Set the SD clock frequency divisor */
58  csrptr->clk_ctl &= SDMC_CLK_CTL_SD_FREQ_HIGH_MASK;
59 
60  /* Internal enable the SD clock */
61  csrptr->clk_ctl |= SDMC_CLK_CTL_INT_CLK_EN;
62 
63  /* Wait for SD clock to be stable */
64  while(!(csrptr->clk_ctl & SDMC_CLK_CTL_INT_CLK_STABLE)) {
65  DELAY(SDMC_CMD_DELAY);
66  }
67 
68  /* Set SD Clock enable */
69  csrptr->clk_ctl |= SDMC_CLK_CTL_SD_CLK_EN;
70 
71  return SDMC_RC_OK;
72 }

88  * sdmcopen - open an SD memory card
89  *------------------------------------------------------------------------
90  */
91 devcall sdmcopen (
92  struct dentry *devptr, /* entry in device switch table */
93  char *name, /* name to open */
94  char *mode /* mode argument */
95  )
96 {
97  volatile struct sdmc_csreg *csrptr; /* address of SD controller's CSR */
98  struct sdmcblk *sdmcptr; /* Pointer to sdmctab entry */
99  uint16 error_sts = 0; /* SDMC command error status */
100  uint32 cmd_arg = 0; /* Value of argument register */
101  byte first_ACMD41 = 1; /* Set for the first ACMD41 */
102 
103  /* Initialize structure pointers */
104  sdmcptr = &sdmctab[devptr->dvminor];
105  csrptr = (struct sdmc_csreg *) devptr->dvcsr;
106 
107  /* Initialize card identifiers */
108  sdmcptr->rca = 0;
109  memset(sdmcptr->cid, 0x00, sizeof(sdmcptr->cid));
110 
111  sdmcptr->cmd8 = 1; /* assume card supports CMD8 */
112  sdmcptr->sdio = 0; /* assume not an SDIO card */
113 
114  /* Set the clock speed */
115  sdmc_set_clock(csrptr);
116 
117  /* Set the bus voltage */
118  sdmc_set_bus_power(csrptr);
119 
120  /* Set the data line timeout value */
121  sdmc_set_dat_timeout(csrptr);
122 
123  /* Issue card reset command (CMD0) */
124  if(sdmc_issue_cmd_sync(csrptr, SDMC_CMD0, 0x00000000, &error_sts, SDMC_CMD_NO_FLAGS) != SDMC_RC_OK) {
125  kprintf("[SDMC] Error in CMD0: %04X %04X", SDMC_CMD0, error_sts);
126  return SYSERR;
127  }
128 
129  /* Issue voltage check command (CMD8) */
130  if(sdmc_issue_cmd_sync(csrptr, SDMC_CMD8, 0x00000101, &error_sts, SDMC_CMD_NO_FLAGS) != SDMC_RC_OK) {
131  kprintf("[SDMC] Error in CMD8: %04X %04X", SDMC_CMD8, error_sts);
132  return SYSERR;
133  }
134  /* Error in CMD8 - card must not support it */
135  if(error_sts & SDMC_ERR_INT_CMD_TIMEOUT_ERR ||
136  csrptr->response0 != 0x00000101) {
137  sdmcptr->cmd8 = 0;
138  }
139 
140  /* Issue inquiry voltage state (ACMD41) */
141  /* To send an application command (ACMD) a CMD55 must first be sent */
142  /* to tell the controller to expect an application command */
143  if(sdmc_issue_cmd_sync(csrptr, SDMC_CMD55, 0x00000000, &error_sts, SDMC_CMD_NO_FLAGS) != SDMC_RC_OK) {
144  kprintf("[SDMC] Error in CMD55: %04X %04X", SDMC_CMD55, error_sts);
145  return SYSERR;
146  }
147  if(sdmc_issue_cmd_sync(csrptr, SDMC_ACMD41, 0x00000000, &error_sts, SDMC_CMD_NO_FLAGS) != SDMC_RC_OK) {
148  kprintf("[SDMC] Error in ACMD41: %04X %04X", SDMC_ACMD41, error_sts);
149  return SYSERR;
150  }
151 
152  /* Issue initialize card command (ACMD41) */
153  cmd_arg = SDMC_OCR_MASK & csrptr->response0; /* Set OCR */
154  if(csrptr->response0 & SDMC_R3_S18A) { /* Set switch to 1.8V is card supports */
155  cmd_arg |= SDMC_ACMD41_S18R;
156  }
157  cmd_arg |= SDMC_ACMD41_XPC | SDMC_ACMD41_HCS; /* Set high capacity support */
158  /* and extended performance control */
159 
160  /* The card initialization command (ACMD41) must be continuously sent until */
161  /* unitialization has completed */
162  do {
163  if(!first_ACMD41) {
164  DELAY(SDMC_CMD_DELAY);
165  }
166 
167  /* To send an application command (ACMD) a CMD55 must first be sent */
168  /* to tell the controller to expect an application command */
169  if(sdmc_issue_cmd_sync(csrptr, SDMC_CMD55, 0x00000000, &error_sts, SDMC_CMD_NO_FLAGS) != SDMC_RC_OK) {
170  kprintf("[SDMC] Error in CMD55: %04X %04X", SDMC_CMD55, error_sts);
171  return SYSERR;
172  }
173  /* Send the card initialization command */
174  if(sdmc_issue_cmd_sync(csrptr, SDMC_ACMD41, cmd_arg, &error_sts, SDMC_CMD_NO_FLAGS) != SDMC_RC_OK) {
175  kprintf("[SDMC] Error in ACMD41: %04X %04X", SDMC_ACMD41, error_sts);
176  return SYSERR;
177  }
178 
179  first_ACMD41 = 0;
180  } while(!(csrptr->response0 & SDMC_R3_BUSY));
181 
182  /* TODO run voltage switch procedure of the card supports 1.8V signaling */
183 
184  /* Retrieve the card's card identifier (CID) */
185  if(sdmc_issue_cmd_sync(csrptr, SDMC_CMD2, 0, &error_sts, SDMC_CMD_NO_FLAGS) != SDMC_RC_OK) {
186  kprintf("[SDMC] Error in CMD2: %04X %04X", SDMC_CMD2, error_sts);
187  return SYSERR;
188  }
189  memcpy(sdmcptr->cid, (char*)&csrptr->response0, 16);
190 
191  /* Retrieve the card's relative card address (RCA) */
192  if(sdmc_issue_cmd_sync(csrptr, SDMC_CMD3, 0, &error_sts, SDMC_CMD_NO_FLAGS) != SDMC_RC_OK) {
193  kprintf("[SDMC] Error in CMD3: %04X %04X", SDMC_CMD3, error_sts);
194  return SYSERR;
195  }
196  sdmcptr->rca = csrptr->response0 & SDMC_R6_RCA_MASK;
197 
198  sdmcptr->cmd_sem = semcreate(0);
199  if((int)sdmcptr->cmd_sem == SYSERR) {
200  return SYSERR;
201  }
202 
203  return devptr->dvnum;
204 }
