5 /*------------------------------------------------------------------------
6  * ptcount - Return the count of messages currently waiting in a port
7  * (a non-negative count K means the port contains K
8  * K messages, including messages from senders that
9  * are blocked waiting to send a message; a count of
10  * negative K means a port is empty and K processes
11  * are blocked waiting to receive a message)
12  *------------------------------------------------------------------------
13  */
14 int32 ptcount(
15  int32 portid /* ID of a port to use */
16  )
17 {
18  intmask mask; /* Saved interrupt mask */
19  int32 count; /* Count of messages available */
20  int32 sndcnt; /* Count of sender semaphore */
21  struct ptentry *ptptr; /* Pointer to port table entry */
22 
23  mask = disable();
24  if ( isbadport(portid) ||
25  (ptptr= &porttab[portid])->ptstate != PT_ALLOC ) {
26  restore(mask);
27  return SYSERR;
28  }
29 
30  /* Get count of messages available */
31 
32  count = semcount(ptptr->ptrsem);
33 
34  /* If messages are waiting, check for blocked senders */
35 
36  if (count >= 0) {
37  sndcnt = semcount(ptptr->ptssem);
38  if (sndcnt < 0) { /* -sndcnt senders blocked */
39  count += -sndcnt;
40  }
41  }
42  restore(mask);
43  return count;
44 }
